class Counter():

    def __init__(self,num):
        self.num = num
        return self.num

    def increase_one(self):
        self.num+=1
        return self.num

    def decrease_one(self):
        self.num-=1
        return self.num

    def get_value(self):
        return self.num

def speedCheck(speed):
    points = 0
    if speed<70:
        print("ok")

    else:
        points = (speed-70)//5

    if points>13:
        print("Licence Suspended")
    else:
        print("Demerit points = ", points)



# 7.Write a function called showNumbers that takes a parameter called limit. It should print all
# the numbers between 0 and limit with a label to identify the even and odd numbers. For example,
# if the limit is 3, it should print:
# 0 EVEN
# 1 ODD
# 2 EVEN
# 3 ODD

def showNumbers(limit):

    for i in range(0, limit):
        if i%2==0:
            print(i," Even")
        else:
            print(i, " Odd")

showNumbers(100)

# 8.Write a function that returns the sum of multiples of 3 and 5 between 0 and limit (parameter).
# For example, if limit is 20, it should return the sum of 3, 5, 6, 9, 10, 12, 15, 18, 20.

def mulSum(limit):
    temp = 0

    for i in range(0,limit+1):
        if i%3==0 or i%5==0:
            temp+=i
    return temp

print(mulSum(20))
